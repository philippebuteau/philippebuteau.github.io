Configuration de serveurs DNS
sous linux

Script bash pour automatiser la configuration de serveurs DNS multiples

RéseautiquebashDNSLinuxBalancement des chargesDNS ForwarderDNS secondaire
Contexte

Conception d'un script paramétrable pour installer de façon complètement automatique un serveur DNS principal et ses serveurs DNS secondaires.

Fonctionnalités
  • Installation de BIND si nécéssaire
  • Configuuration du serveur principal en fonctions des paramètres définis
  • Envoi et exécution automatique du script vers les serveurs secondaires via SSH
  • Supporte les zones directes, inverses, forwarders et slaves
Étape 1: Définition des variables (paramètres)

Les variables suivantes sont configurées en haut du script. Elles sont divisées en blocs et chaque bloc à une switch ON/OFF en fonction de la configuration désirée.


# Text Variables
BOLD=$'\e[1m'
RESET=$'\e[0m'
RED='\033[0;31m'
GREEN='\033[0;32m'

# General Variables
DATE=$(date +%Y%m%d%M)
ADMIN="admin"
NAMED_CONF="/etc/named.conf"

# Options Block Variables
SWITCH_OPTIONS="ON"
FORWARD_OPTION="forward first"
FORWARDERS="8.8.8.8; 1.1.1.1;"
OPEN_PORTS="any;"
ALLOWED_FROM="any;"

# Master Zone Variables
SWITCH_MASTER="ON"
SRV="R1"
DOMAIN="buteau.tld"
DNS_IP="192.168.26.1"
PTR_NETWORK="26.168.192"
PTR_ADDRESS="1"
ZONE_FILE="/var/named/$DOMAIN.db"
ZONE_FILE_REVERSED="/var/named/$PTR_NETWORK.db"

# Forwarder Zone Variables (To configure a DNS forwarder on a distant machine)
SWITCH_FORWARDER="OFF"
SWITCH_EXEC_FORWARDER="OFF"					# Should always be OFF, will be switched automatically once on distant computer
FORWARDER_SRV="A"						# For SSH
FORWARDER_USER="admin"						# For SSH
FORWARDER_IP="192.168.26.20"					# For SSH
FORWARDER_DOMAIN="buteau.tld"
FORWARDER_OPTION="forward only"
FORWARDER_DESTINATIONS="192.168.26.1;"

# Slave Zone Variables (To configure a DNS slave on a distant machine)
SWITCH_SLAVE="ON"
SWITCH_EXEC_SLAVE="OFF"						# Should always be OFF, will be switched automatically once on distant computer
SLAVE_SRV="A"							# For SSH
SLAVE_USER="admin"						# For SSH
SLAVE_IP="192.168.26.20"					# For SSH
SLAVE_DOMAIN="buteau.tld"
MASTERS="192.168.26.1;"
SLAVE_PTR_NETWORK="26.168.192"

Fonctions: Installation de BIND

# Install package
install_bind() {
	echo -e "\n${BOLD}BIND installation check:${RESET}"
	if [[ $(rpm -q bind) == "package bind is not installed" ]]; then
		echo -e "${BOLD}BIND is not installed on this machine; proceeding to installation..${RESET}"
		dnf install -y bind --quiet
	else
		echo -e "BIND is installed on this machine: $(rpm -q bind)"
	fi
}
Fonctions: Configuration du block option du fichier named.conf

# Forwarders (options block)
config_opt_forwarders() {
	echo -e "\n${BOLD}Searching configuration file for forwarders${RESET}"
	if [[ $(sed -n '/^\s*forwarders\s.*;/p' $NAMED_CONF)  == "" ]]; then
		echo -e "Adding forwarders to configuration"
		sed -i '/options\s*{/a\        forwarders {'"$FORWARDERS"'};' $NAMED_CONF
	else
		echo -e "Forwarders option was already configured (updated)"
		sed -i 's/^\s*forwarders\s{.*/        forwarders {'"$FORWARDERS"'};/' $NAMED_CONF
	fi
}

# Forward (options block)
config_opt_forward() {
	echo -e "\n${BOLD}Searching configuration file for forward option${RESET}"
	if [[ $(sed -n '/^\s*forward\s.*;/p' $NAMED_CONF)  == "" ]]; then
		echo -e "Adding forward option to configuration"
		sed -i '/options\s*{/a\        '"$FORWARD_OPTION"';' $NAMED_CONF
	else
		echo -e "Forward option was already configured (updated)"
		sed -i 's/^\s*forward\s.*/        '"$FORWARD_OPTION"';/' $NAMED_CONF
	fi
}
# Listen-on ports
config_opt_listen_on() {
	echo -e "\n${BOLD}Configuring: listen-on${RESET}\n-> $OPEN_PORTS"
	sed -i 's/^\s*listen-on port.*/	listen-on port 53 { '"$OPEN_PORTS"' };/' $NAMED_CONF
}
# Allow queries from
config_opt_queries_from() {
	echo -e "\n${BOLD}Configuring: allow-query${RESET}\n-> $ALLOWED_FROM"
	sed -i 's/^\s*allow-query.*/	allow-query { '"$ALLOWED_FROM"' };/' $NAMED_CONF
}
Fonctions: Configuration des zones directes et inverses

# Add forward zone
config_zone_forward() {
	echo -e "\n${BOLD}Declaring zone: ${DOMAIN}${RESET}"
	if ! grep -q "zone \"$DOMAIN\"" "$NAMED_CONF"; then
		cat >> "$NAMED_CONF" --EOF
zone "$DOMAIN" IN {
	type master;
	file "$DOMAIN.db";
	allow-update { none; };
};
EOF
		echo -e "Declared zone: $DOMAIN "
	else
		echo -e "$DOMAIN zone was already declared"
	fi
}

# Add reversed zone
config_zone_reversed() {
	if ! grep -q "zone \"$PTR_NETWORK.in.addr.arpa\"" "$NAMED_CONF"; then
		cat >> "$NAMED_CONF" --EOF
zone "$PTR_NETWORK.in-addr.arpa" IN {
	type master;
	file "$PTR_NETWORK.db";
};
EOF
		echo -e "Declared reversed zone: $PTR_NETWORK"
	else
		echo -e "$PTR_NETWORK zone was already declared"
	fi
}
Fonctions: Configuration des fichiers de zones

# Adding zone file
config_zonefile_forward() {
	if [ ! -f "$ZONE_FILE" ]; then
		cat > "$ZONE_FILE" --EOF
\$TTL 86400
@   IN  SOA     $SRV.$DOMAIN. $ADMIN.$DOMAIN. (
        $DATE	; Serial
        3600		; Refresh
        1800		; Retry
        604800		; Expire
        86400 )		; Minimum TTL

@		IN  NS      $SRV.$DOMAIN.
$SRV		IN  A       $DNS_IP
mail		IN  A       $DNS_IP
www		IN  A       $DNS_IP

@		IN  MX 10   mail.$DOMAIN.
EOF
		chown named:named "$ZONE_FILE"
		chmod 640 "$ZONE_FILE"

		echo "Created zone file: $ZONE_FILE"
	else
		echo "$ZONE_FILE already exists"
	fi
}

# Adding reversed zone file
config_zonefile_reversed() {
	if [ ! -f "$ZONE_FILE_REVERSED" ]; then
		cat > "$ZONE_FILE_REVERSED" --EOF
\$TTL 86400
@   IN  SOA     $SRV.$DOMAIN. $ADMIN.$DOMAIN. (
        $DATE	; Serial
        3600		; Refresh
        1800		; Retry
        604800		; Expire
        86400 )		; Minimum TTL

@		IN  NS	$SRV.$DOMAIN.
$PTR_ADDRESS		IN	PTR	$SRV.$DOMAIN.
EOF
		chown named:named "$ZONE_FILE_REVERSED"
		chmod 640 "$ZONE_FILE_REVERSED"

		echo "Created zone file: $ZONE_FILE_REVERSED"
	else
		echo "$ZONE_FILE_REVERSED already exists"
	fi
}
Fonctions: Configuration d'une zone slave

# Add slave zone
config_zone_slave() {
	echo -e "\n${BOLD}Declaring slave zone: ${SLAVE_DOMAIN}${RESET}"
	if ! grep -q "zone \"$SLAVE_DOMAIN\"" "$NAMED_CONF"; then
		cat >> "$NAMED_CONF" --EOF

zone "$SLAVE_DOMAIN" IN {
	type slave;
	file "slaves/$DOMAIN.db";
	masters { $MASTERS };
};

zone "$SLAVE_PTR_NETWORK.in-addr.arpa" IN {
	type slave;
	file "slaves/$SLAVE_PTR_NETWORK.db";
	masters { $MASTERS };
};
EOF
		echo -e "Declared slave zone: $SLAVE_DOMAIN "
	else
		echo -e "$SLAVE_DOMAIN slave zone was already declared"
	fi
}
Fonctions: Configuration d'une zone forwarder

# Add forwarder zone
config_zone_forwarder() {
	echo -e "\n${BOLD}Declaring forwarder zone: ${FORWARDER_DOMAIN} ON ${FORWARDER_SRV}.${FORWARDER_IP}${RESET}"
	if ! grep -q "zone \"$FORWARDER_DOMAIN\"" "$NAMED_CONF"; then
		cat >> "$NAMED_CONF" --EOF

zone "$FORWARDER_DOMAIN" IN {
	type forward;
	$FORWARDER_OPTION;
	forwarders { $FORWARDER_DESTINATIONS };
};
EOF
		echo -e "Declared forwarder zone: $FORWARDER_DOMAIN "
	else
		echo -e "$FORWARDER_DOMAIN forwarder zone was already declared"
	fi
}
Fonctions: Activation et vérification du service

config_check() {
	echo -e "\n${BOLD}=== Verification ===${RESET}"
	error1=$(named-checkconf 2>&1)
	exit_code1=$?
	error2=$(named-checkzone "$DOMAIN" "$ZONE_FILE"	 2>&1)
	exit_code2=$?
	if [[ "$exit_code1" -ne 0 ]]; then
		echo -e "${RED}ERROR(S):\n/etc/named.conf   -> $error1\n${BOLD}EXITING EARLY${RESET}"
		exit
	elif [[ "$exit_code2" -ne 0 ]]; then
		echo -e "${RED}WARNING(S):\n/var/named (zone) -> $error2\n${RESET}"
	fi
}

config_resolv() {
	chattr -i /etc/resolv.conf
	cat > /etc/resolv.conf --EOF
search localdomain $DOMAIN
nameserver $DNS_IP
EOF
	chattr +i /etc/resolv.conf
	echo -e "\nnslookup"
	nslookup google.ca	
	nslookup $DNS_IP
}
Exécution

################################## Execution ##################################
# install package
install_bind

# options
if [[ "$SWITCH_OPTIONS" == "ON" ]]; then
	config_opt_forward
	config_opt_forwarders
	config_opt_listen_on
	config_opt_queries_from
fi

# master
if [[ "$SWITCH_MASTER" == "ON" ]]; then
	config_zone_forward
	config_zonefile_forward
	config_zone_reversed
	config_zonefile_reversed
fi

# Setup another machine as forwarder
if [[ "$SWITCH_FORWARDER" == "ON" ]]; then
	if [[ "$SWITCH_EXEC_FORWARDER" == "OFF" ]]; then
		# Commands on distant computer
		# Remaps some of the variables to the proper values and executes this script
		echo -e "${BOLD}\n####### Switching to distant machine #######\n${RESET}"
		scp ./dnsSetup.sh $FORWARDER_USER@$FORWARDER_IP:~/
		scp ./dnsUnSetup.sh $FORWARDER_USER@$FORWARDER_IP:~/
		ssh $FORWARDER_USER@$FORWARDER_IP "
			sudo ~/dnsUnSetup.sh
			sed -i 's/^SRV=.*/SRV=\"$FORWARDER_SRV\"/' ~/dnsSetup.sh; 
			sed -i 's/^DOMAIN=.*/DOMAIN=\"$FORWARDER_DOMAIN\"/' ~/dnsSetup.sh; 
			sed -i 's/^DNS_IP=.*/DNS_IP=\"$FORWARDER_IP\"/' ~/dnsSetup.sh; 
			sed -i 's/^SWITCH_MASTER=\"ON\"/SWITCH_MASTER=\"OFF\"/' ~/dnsSetup.sh; 
			sed -i 's/^SWITCH_EXEC_FORWARDER=\"OFF\"/SWITCH_EXEC_FORWARDER=\"ON\"/' ~/dnsSetup.sh;
			sudo ~/dnsSetup.sh
			"
	else # This will only execute from the distant machine
		sed -i 's/^\s*dnssec-validation yes;/	dnssec-validation no;/' $NAMED_CONF; 
		config_zone_forwarder
		systemctl enable named
		systemctl restart named
		config_resolv
		echo -e "${BOLD}\n####### Switching back to local machine #######\n${RESET}"
		exit
	fi
fi

# Setup another machine as slave for zone transfer
if [[ "$SWITCH_SLAVE" == "ON" ]]; then
	if [[ "$SWITCH_EXEC_SLAVE" == "OFF" ]]; then
		# Configure master zone for transfer
		sed -i "/^zone $DOMAIN IN {/a allow-transfer { $SLAVE_IP; };\nalso-notify { $SLAVE_IP; };" "$NAMED_CONF"

		# Commands on distant computer
		# Remaps some of the variables to the proper values for the distant machine and executes this script
		echo -e "${BOLD}\n####### Switching to distant machine #######\n${RESET}"
		scp ./dnsSetup.sh $SLAVE_USER@$SLAVE_IP:~/
		scp ./dnsUnSetup.sh $SLAVE_USER@$SLAVE_IP:~/
		ssh $SLAVE_USER@$SLAVE_IP "
			sudo ~/dnsUnSetup.sh
			sed -i 's/^SRV=.*/SRV=\"$SLAVE_SRV\"/' ~/dnsSetup.sh;
			sed -i 's/^DOMAIN=.*/DOMAIN=\"$SLAVE_DOMAIN\"/' ~/dnsSetup.sh;
			sed -i 's/^DNS_IP=.*/DNS_IP=\"$SLAVE_IP\"/' ~/dnsSetup.sh;
			sed -i 's/^SWITCH_MASTER=\"ON\"/SWITCH_MASTER=\"OFF\"/' ~/dnsSetup.sh;
			sed -i 's/^SWITCH_EXEC_SLAVE=\"OFF\"/SWITCH_EXEC_SLAVE=\"ON\"/' ~/dnsSetup.sh;
			sudo ~/dnsSetup.sh
			"
	else # This will only execute from the distant machine
		sed -i 's/^\s*dnssec-validation yes;/	dnssec-validation no;/' $NAMED_CONF;
		config_zone_slave
		systemctl enable named
		systemctl restart named
		config_resolv
		echo -e "\n${GREEN}${BOLD}Zone transfer logs: /var/named/data/named.run${RESET}"
		echo -e "\n${BOLD}Commands to push an update on the slave:${RESET}"
		echo -e "${BOLD}  On master: rndc reload${RESET}"
		echo -e "${BOLD}  On slave : rndc retransfer DOMAIN.TLD${RESET}"
		echo -e "${BOLD}  On slave : rndc refresh DOMAIN.TLD${RESET}"
		echo -e "\n${BOLD}\n####### Switching back to local machine #######\n${RESET}"
		exit
	fi
fi

# finalize
config_check
systemctl enable named
systemctl restart named
config_resolv

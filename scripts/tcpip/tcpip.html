Configuration TCP/IP
sous linux

Script bash pour automatiser la configuration des interfaces réseau sous linux/

RéseautiquebashVLANBONDAdressage
Contexte

Conception d'un script paramétrable pour configurer les interfaces réseaux sous linux

Fonctionnalités
  • Configuration IPv4 de base
  • Configuration d'interfaces en VLAN
  • Configuration d'interface en mode BOND (Plusieurs interfaces logiquement agrégées)
  • Affichage de la configuration actuelle
  • Supprimer la configuration d'une interface
  • Supprimer la configuration de toutes les interfaces
  • Configurer automatiquement toutes les interfaces depuis mariadb
Fonctions: Configuration d'un profil ethernet, vlan ou bonded


function setup-ethernet-profile() {
	ACTIVE=$(nmcli connection show | sed -n "/${INTERFACES[$i]}/ p")
	if [[ -z $ACTIVE ]]; then
		nmcli connection add type ethernet ifname ${INTERFACES[$i]} con-name ${INTERFACES[$i]}
	fi
}
function setup-vlan-profile() {
	if [[ -z $AUTO ]]; then
		echo -e "${BOLD}Enter VLAN id (Number): ${RESET}"
		read VLANID
		echo -e "${BOLD}Enter VLAN ip address (xxx.xxx.xxx.xxx/xx): ${RESET}"
		read IP
		echo -e "${BOLD}Enter VLAN gateway (Press ENTER to skip): ${RESET}"
		read GW
		echo -e "${BOLD}Enter VLAN DNS (Press ENTER to skip): ${RESET}"
		read DNS
	fi
	if [[ -n $GW && $GW != "NULL" ]]; then
		nmcli connection add type vlan con-name ${INTERFACES[$i]}.${VLANID} dev ${INTERFACES[$i]} id $VLANID ip4 $IP gw4 $GW	
	else
		nmcli connection add type vlan con-name ${INTERFACES[$i]}.${VLANID} dev ${INTERFACES[$i]} id $VLANID ip4 $IP
	fi

	if [[ -n $DNS && $DNS != "NULL" ]]; then
		nmcli connection modify "${INTERFACES[$i]}.${VLANID}" ipv4.dns $DNS
	fi
	INTERFACES[$i]=${INTERFACES[$i]}.${VLANID}
}
function setup-bonded-profile() {
	if [[ -z $AUTO ]]; then
		echo -e "${BOLD}Enter bond name: ${RESET}"
		read NAME
		echo "Bond modes: balance-rr, active-backup, 802.3ad, balance-xor, broadcast, balance-tlb, balance-alb"
		echo -e "${BOLD}Enter the mode of the bond: ${RESET}"
		read MODE
		echo -e "${BOLD}Enter the number of slaves for this bond: ${RESET}"
		read SLAVES
	fi
	if [[ -z $SLAVES ]]; then
		SLAVES=1
	fi
	nmcli connection add type bond con-name $NAME ifname $NAME mode $MODE
	SLAVENAME="${INTERFACES[$i]}.$NAME"
	nmcli connection add type bond-slave con-name $SLAVENAME-slave$NUM ifname ${INTERFACES[$i]} master $NAME
	INTERFACES[$i]=$NAME
}
function remove-profile() {
	nmcli
	echo ""
	echo -e "${BOLD}Enter interface to remove: ${RESET}"
	read NAME
	nmcli connection delete $NAME
	menu
}
function wipe-profiles() {
	mapfile -t UUIDS - -(nmcli connection show | gawk -F" " '$1 != "lo" && $1 != "NAME"  {print $2}')
	for UUID in ${UUIDS[@]}; do
		nmcli connection delete $UUID
	done
}
Fonctions: Configuration TCP/IP de base

Méthode auto/statique, adresse IP, passerelle par défaut, serveurs DNS, nom de l'interface


function query-method() {
	if [[ -z $AUTO ]]; then
		echo -e "${BOLD}Enter ipv4 method (static ; auto): ${RESET}"
		read METHOD
	fi
}
function setup-method() {
	if [[ $METHOD == "static" || $METHOD == "auto" ]]; then
		nmcli connection modify ${INTERFACES[$i]} ipv4.method $METHOD
	fi
}
function setup-address() {
	if [[ -z $AUTO ]]; then
		echo -e "${BOLD}Enter ipv4 address/mask (ENTER to skip): ${RESET}"
		read IP
	fi
	if [[ -n $IP ]]; then
		nmcli connection modify ${INTERFACES[$i]} ipv4.address ${IP}
	fi	
}
function setup-gateway() {
	if [[ -z $AUTO ]]; then	
		echo -e "${BOLD}Enter default gateway (ENTER to skip): ${RESET}"
		read GW
	fi
	if [[ -n $GW && $GW != "NULL" ]]; then
		nmcli connection modify ${INTERFACES[$i]} ipv4.gateway $GW
	fi	
}
function setup-dns() {
	if [[ -z $AUTO ]]; then	
		echo -e "${BOLD}Enter dns address (Comma separated list) (ENTER to skip): ${RESET}"
		read DNS
	fi
	if [[ -n $DNS && $DNS != "NULL" ]]; then
		nmcli connection modify ${INTERFACES[$i]} ipv4.dns $DNS
	fi	
}
function append-dns() {
	CURRENT=$(nmcli -g ipv4.dns connection show ${INTERFACES[$i]})
	echo -e "${BOLD}Current dns servers: ${RESET}\n$CURRENT"
	echo ""
	echo -e "${BOLD}Enter dns address (Comma separated list) (ENTER to skip): ${RESET}"
	read DNS
	if [[ -n $DNS ]]; then
		nmcli connection modify ${INTERFACES[$i]} ipv4.dns "${CURRENT},${DNS}"
	fi	
}
function activate-interface(){
	nmcli connection up ${INTERFACES[$i]}
}
function change-device-name(){
	if [[ -z $AUTO ]]; then
		echo -e "${BOLD}Enter the new name for this device (ENTER to skip): ${RESET}"
		read NAME
	fi
	if [[ -n $NAME ]]; then
		nmcli connection modify ${INTERFACES[$i]} con-name $NAME
	fi
	INTERFACES[$i]=$NAME
}
Fonctions: Récupération des informations sur les interfaces

Ces informations sont utilisées par les autres fonctions, elles permettent de choisir les interfaces à configurer


function get-info() {
	mapfile -t INTERFACES - -(nmcli | grep connected | gawk -F" " '$4 != "lo" && $4 != "to" && NF >= 4 { print $4 }')
	mapfile -t -O "${#INTERFACES[@]}" INTERFACES - -(nmcli | grep disconnected | gawk -F":" '$1 != "lo" { print $1 }')
	mapfile -t ADDRESSES - -(nmcli | grep inet4 | gawk -F" " '$2 != "127.0.0.1/8" { print $2 }' | gawk -F"/" '{ print $1 }')
	mapfile -t MASKS - -(nmcli | grep inet4 | gawk -F" " '$2 != "127.0.0.1/8" { print $2 }' | gawk -F"/" '{ print $2 }')
	echo ""
	echo -e "${BOLD}Connected interfaces: ${RESET}"
	# Display
	for i in "${!INTERFACES[@]}"; do
		echo "${i}) ${INTERFACES[$i]} : ${ADDRESSES[$i]}/${MASKS[$i]}"
	done
	echo ""
}
Fonctions: Séquences d'action

Séquences de fonctions pour accomplir différentes tâches


function action-sequence-setIP() {
	setup-ethernet-profile
	query-method
	if [[ $METHOD == "static" ]]; then
		setup-address
		setup-gateway
		setup-dns
		setup-method
	elif [[ $METHOD == "auto" ]]; then
		setup-method
	fi
	activate-interface
}
function action-sequence-vlan() {
	setup-vlan-profile
	activate-interface
}
function action-sequence-name() {
	change-device-name
	activate-interface
}
function action-sequence-bond() {
	NUM=1
	setup-bonded-profile
	action-sequence-setIP
	for NUM in $(seq 2 $SLAVES); do
		process addto-bonded-profile
	done
}
Fonctions: Process

Ces fonctions sont conçue pour itérerer sur les interfaces disponibles. Elles attendent l'une des séquences d'actions précédentes comme argument

Le mode normal est fait pour opérer sur une interface interactivement sélectionnée par l'utilisateur. Le mode automatique itère sur toutes les interfaces


function process() {
	get-info
	echo -e "${BOLD}Select an interface to modify: ${RESET}"
	read CHOICE
	for i in "${!INTERFACES[@]}"; do
		if [[ "$i" -eq "$CHOICE" ]]; then
			# Run parameter function $1
			$1
		fi
	done
}
function process-auto() {
	for i in "${INTERFACES[@]}"; do
		if [[ "$i" == "$KEY" ]]; then
			INTERFACES[$i]=$KEY
			# Run parameter function $1
			$1
		fi
	done
}
Fonctions: Interaction avec mariadb

Ces fonctions permettent l'automatisation de la configuration via une base de données


function extract-from-database() {
	wipe-profiles
	get-info
	mapfile -t KEYS - -(mysql networkInterfaces -B -N -e "SELECT device FROM NIC")
	# Regular ehternet interface
	for KEY in "${KEYS[@]}"; do
		AUTO=1
		IP=$(mysql -e "USE networkInterfaces; SELECT GROUP_CONCAT(ipv4) FROM address
			JOIN NIC ON address.deviceID = NIC.deviceID
			WHERE NIC.device = '$KEY';" -B -N)
		GW=$(mysql -e "USE networkInterfaces; SELECT GROUP_CONCAT(gateway) FROM address
			JOIN NIC ON address.deviceID = NIC.deviceID
			WHERE NIC.device = '$KEY';" -B -N)
		DNS=$(mysql -e "USE networkInterfaces; SELECT GROUP_CONCAT(dns) FROM address
			JOIN NIC ON address.deviceID = NIC.deviceID
			WHERE NIC.device = '$KEY';" -B -N)
		METHOD=$(mysql -e "USE networkInterfaces; SELECT method FROM NIC WHERE device = '$KEY';" -B -N)
		PROFILE=$(mysql -e "USE networkInterfaces; SELECT profile FROM NIC WHERE device = '$KEY';" -B -N)
		mapfile -t VLANIDS - -(mysql -e "USE networkInterfaces; SELECT vlanTAG FROM vlan
										JOIN NIC ON vlan.deviceID = NIC.deviceID
										WHERE NIC.device = '$KEY';" -B -N)
		if [[ "$PROFILE" == "ethernet" ]]; then
			echo -e "${BOLD}$KEY:$IP:$GW:$DNS:$METHOD:$PROFILE${RESET}"
			process-auto action-sequence-setIP		
		# VLAN interface
		elif [[ "$PROFILE" == "vlan" ]]; then
			for VLANID in ${VLANIDS[@]}; do
				IP=$(mysql -e "USE networkInterfaces; SELECT address.ipv4 FROM NIC
					JOIN vlan ON NIC.deviceID = vlan.deviceID
					JOIN address ON address.addressID = vlan.addressID
					WHERE NIC.device = '$KEY' AND vlan.vlanTAG = '$VLANID';" -B -N)
				GW=$(mysql -e "USE networkInterfaces; SELECT address.gateway FROM NIC
					JOIN vlan ON NIC.deviceID = vlan.deviceID
					JOIN address ON address.addressID = vlan.addressID
					WHERE NIC.device = '$KEY' AND vlan.vlanTAG = '$VLANID';" -B -N)
				DNS=$(mysql -e "USE networkInterfaces; SELECT address.dns FROM NIC
					JOIN vlan ON NIC.deviceID = vlan.deviceID
					JOIN address ON address.addressID = vlan.addressID
					WHERE NIC.device = '$KEY' AND vlan.vlanTAG = '$VLANID';" -B -N)
				METHOD=$(mysql -e "USE networkInterfaces; SELECT method FROM NIC WHERE device = '$KEY';" -B -N)
				echo -e "${BOLD}$KEY:$IP:$GW:$DNS:$METHOD:$PROFILE:$VLANID${RESET}"
				process-auto action-sequence-vlan
			done
		# Bonded interfaces
		elif [[ "$PROFILE" == "bond" ]]; then
			IP=$(mysql -e "USE networkInterfaces; SELECT address.ipv4 FROM NIC
				JOIN bond ON NIC.deviceID = bond.deviceID
				JOIN address ON address.addressID = bond.addressID
				WHERE NIC.device = '$KEY';" -B -N)
			GW=$(mysql -e "USE networkInterfaces; SELECT address.gateway FROM NIC
				JOIN bond ON NIC.deviceID = bond.deviceID
				JOIN address ON address.addressID = bond.addressID
				WHERE NIC.device = '$KEY';" -B -N)
			DNS=$(mysql -e "USE networkInterfaces; SELECT address.dns FROM NIC
				JOIN bond ON NIC.deviceID = bond.deviceID
				JOIN address ON address.addressID = bond.addressID
				WHERE NIC.device = '$KEY';" -B -N)
			METHOD=$(mysql -e "USE networkInterfaces; SELECT method FROM NIC WHERE device = '$KEY';" -B -N)
			NAME=$(mysql -e "USE networkInterfaces; SELECT bondTAG FROM bond
				JOIN NIC ON NIC.deviceID = bond.deviceID
				WHERE NIC.device = '$KEY';" -B -N)
			ISMASTER=$(mysql -e "USE networkInterfaces; SELECT ismaster FROM bond
				JOIN NIC ON NIC.deviceID = bond.deviceID
				WHERE NIC.device = '$KEY';" -B -N)
			NUM=$(mysql -e "USE networkInterfaces; SELECT slaveNUM FROM bond
				JOIN NIC ON NIC.deviceID = bond.deviceID
				WHERE NIC.device = '$KEY';" -B -N)
			MODE=$(mysql -e "USE networkInterfaces; SELECT mode FROM bond
				JOIN NIC ON NIC.deviceID = bond.deviceID
				WHERE NIC.device = '$KEY';" -B -N)
			INTERFACES[$i]=$KEY
			echo -e "${BOLD}$KEY:$NAME:$MODE:$ISMASTER:$SLAVENUM:$IP:$GW:$DNS:$METHOD:$PROFILE:$VLANID${RESET}"
			if [[ $ISMASTER == "TRUE" ]]; then
				setup-bonded-profile
				action-sequence-setIP
			else
				addto-bonded-profile
			fi
		fi
		unset AUTO
	done
	echo -e "${BOLD}Press ENTER to continue${RESET}"
	read "VOID"
	menu
}
function display_database() {
	echo ""
	echo -e "${BOLD}Table: NIC${RESET}"
	mysql -e "USE networkInterfaces; SELECT * FROM NIC;"
	echo -e "${BOLD}Table: address${RESET}"
	mysql -e "USE networkInterfaces; SELECT * FROM address;"
	echo -e "${BOLD}Table: vlan${RESET}"
	mysql -e "USE networkInterfaces; SELECT * FROM vlan;"
	echo -e "${BOLD}Table: bond${RESET}"
	mysql -e "USE networkInterfaces; SELECT * FROM bond;"
	echo ""
}
Fonctions: Menu

Menu interactif pour l'exécution


function menu() {
	clear
	echo -e "${BOLD}	Network Manager	${RESET}"
	cat -- EOF
0)....Display interfaces
1)....Change device name
2)....Setup networking for an interface
3)....Append DNS server
4)....Remove profile (Delete interface configuration)
5)....Remove all profiles
6)....Setup VLAN on interface
7)....Setup bonded interfaces
8)....Automated from database info
9)....Display infos from database

Q)....Quit	
EOF
	echo -e "${BOLD}Choose an action from the menu: ${RESET}"
	read INPUT
	case $INPUT in
		0) clear;get-info;echo -e "${BOLD}Press ENTER to continue${RESET}";read VOID;menu;;
		1) clear;process action-sequence-name; unset INPUT;echo -e "${BOLD}Press ENTER to continue${RESET}";read VOID;menu;;
		2) clear;process action-sequence-setIP; unset INPUT;echo -e "${BOLD}Press ENTER to continue${RESET}";read VOID;menu;;
		3) clear;process append-dns; unset INPUT;echo -e "${BOLD}Press ENTER to continue${RESET}";read VOID;menu;;
		4) clear;remove-profile;unset INPUT;;
		5) clear;wipe-profiles;unset INPUT;echo -e "${BOLD}Press ENTER to continue${RESET}";read VOID;menu;;
		6) clear;process action-sequence-vlan; unset INPUT;echo -e "${BOLD}Press ENTER to continue${RESET}";read VOID;menu;;
		7) clear;process action-sequence-bond; unset INPUT;echo -e "${BOLD}Press ENTER to continue${RESET}";read VOID;menu;;
		8) clear;extract-from-database;unset INPUT;;
		9) clear;display_database;unset INPUT;echo -e "${BOLD}Press ENTER to continue${RESET}";read VOID;menu;;
		[qQ]) exit;;
	esac
}
